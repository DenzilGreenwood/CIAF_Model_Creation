from ciaf.supply_chain import AISupplyChainGovernance
from ciaf.vendor import VendorRiskManagement
from ciaf.provenance import ModelProvenanceTracker

class AISupplyChainFramework:
    def __init__(self):
        self.governance = AISupplyChainGovernance()
        self.vendor_mgmt = VendorRiskManagement()
        self.provenance = ModelProvenanceTracker()
        
    def track_model_lifecycle(self, model_development_request):
        """Complete model lifecycle tracking with supply chain governance"""
        
        # Initialize provenance tracking
        provenance_id = self.provenance.initialize_tracking(
            model_request=model_development_request,
            timestamp=datetime.utcnow()
        )
        
        # Data source verification
        data_sources = self.governance.verify_data_sources(
            requested_sources=model_development_request.data_sources,
            licensing_requirements=model_development_request.licensing_needs,
            privacy_constraints=model_development_request.privacy_requirements
        )
        
        # Vendor assessment for external services
        if model_development_request.external_vendors:
            vendor_assessment = self.vendor_mgmt.assess_vendors(
                vendors=model_development_request.external_vendors,
                risk_tolerance=model_development_request.risk_level,
                compliance_requirements=['ISO27001', 'SOC2', 'GDPR']
            )
            
            # Update provenance with vendor information
            self.provenance.record_vendor_involvement(
                provenance_id=provenance_id,
                vendor_assessment=vendor_assessment
            )
        
        # Model development tracking
        development_tracker = self.governance.track_model_development(
            provenance_id=provenance_id,
            data_sources=data_sources,
            development_environment=model_development_request.environment,
            version_control=model_development_request.version_control
        )
        
        # Cryptographic verification setup
        crypto_framework = self.governance.setup_cryptographic_verification(
            provenance_id=provenance_id,
            verification_requirements=['model_integrity', 'supply_chain_signature']
        )
        
        return AISupplyChainTracker(
            provenance_id=provenance_id,
            data_sources=data_sources,
            vendor_assessment=vendor_assessment if model_development_request.external_vendors else None,
            development_tracker=development_tracker,
            crypto_framework=crypto_framework
        )
        
    def verify_supply_chain_integrity(self, model_id):
        """Comprehensive supply chain integrity verification"""
        
        # Retrieve complete provenance record
        provenance_record = self.provenance.get_complete_record(model_id)
        
        # Verify cryptographic signatures
        signature_verification = self.governance.verify_signatures(
            provenance_record=provenance_record
        )
        
        # Vendor compliance check
        vendor_compliance = self.vendor_mgmt.verify_ongoing_compliance(
            vendors=provenance_record.involved_vendors,
            compliance_date=datetime.utcnow()
        )
        
        # Data source licensing verification
        licensing_verification = self.governance.verify_data_licensing(
            data_sources=provenance_record.data_sources,
            usage_context=provenance_record.intended_use
        )
        
        # Generate compliance report
        compliance_report = self.governance.generate_supply_chain_report(
            model_id=model_id,
            signature_verification=signature_verification,
            vendor_compliance=vendor_compliance,
            licensing_verification=licensing_verification,
            eu_ai_act_assessment=self.assess_eu_ai_act_compliance(provenance_record)
        )
        
        return compliance_report
