from ciaf.defense import DefenseAIGovernanceFramework
from ciaf.weapons import AutonomousWeaponsOversight
from ciaf.military import HumanInTheLoopController

class DefenseAISystem:
    def __init__(self, classification_level='SECRET'):
        self.governance = DefenseAIGovernanceFramework()
        self.weapons_oversight = AutonomousWeaponsOversight()
        self.human_controller = HumanInTheLoopController()
        self.classification = classification_level
        
    def deploy_autonomous_weapon_system(self, weapon_system, rules_of_engagement):
        """Deploy autonomous weapon system with comprehensive oversight"""
        
        # Human oversight requirement validation
        human_oversight = self.human_controller.validate_human_oversight_requirements(
            weapon_system=weapon_system,
            lethality_level=weapon_system.lethality_classification
        )
        
        if not human_oversight.sufficient:
            return self.reject_deployment("Insufficient human oversight")
        
        # Rules of engagement compliance
        roe_compliance = self.weapons_oversight.validate_rules_of_engagement(
            weapon_system=weapon_system,
            rules_of_engagement=rules_of_engagement,
            operational_context=weapon_system.deployment_context
        )
        
        # Ethical review for lethal systems
        if weapon_system.is_lethal:
            ethical_review = self.governance.conduct_lethal_system_review(
                weapon_system=weapon_system,
                review_board=['ethics_officer', 'legal_counsel', 'operational_commander']
            )
            
            if not ethical_review.approved:
                return self.reject_deployment("Ethical review failed")
        
        # Civilian protection assessment
        civilian_protection = self.weapons_oversight.assess_civilian_risk(
            weapon_system=weapon_system,
            operational_area=weapon_system.deployment_area,
            rules_of_engagement=rules_of_engagement
        )
        
        # Deploy with mandatory human control
        deployment = self.governance.deploy_with_human_control(
            weapon_system=weapon_system,
            human_oversight=human_oversight,
            roe_compliance=roe_compliance,
            civilian_protection=civilian_protection,
            monitoring_requirements=['real_time', 'audit_trail', 'intervention_capability']
        )
        
        # Continuous monitoring and audit
        monitoring_system = self.governance.deploy_autonomous_weapon_monitoring(
            deployment=deployment,
            audit_frequency='continuous',
            human_intervention_threshold=0.95  # 95% confidence required for autonomous action
        )
        
        return AutonomousWeaponDeployment(
            weapon_system=weapon_system,
            human_oversight=human_oversight,
            ethical_approval=ethical_review if weapon_system.is_lethal else None,
            roe_compliance=roe_compliance,
            civilian_protection=civilian_protection,
            deployment=deployment,
            monitoring_system=monitoring_system,
            deployment_classification=self.classification
        )
        
    def process_target_engagement_request(self, target_data, weapon_system):
        """Process target engagement with human authorization requirement"""
        
        # Target identification and validation
        target_validation = self.weapons_oversight.validate_target(
            target_data=target_data,
            weapon_system=weapon_system,
            rules_of_engagement=weapon_system.current_roe
        )
        
        # Civilian casualty assessment
        casualty_assessment = self.weapons_oversight.assess_casualty_risk(
            target_data=target_data,
            weapon_system=weapon_system,
            area_intelligence=self.get_area_intelligence(target_data.location)
        )
        
        # Mandatory human authorization for lethal action
        if weapon_system.is_lethal and target_validation.valid:
            human_authorization = self.human_controller.request_lethal_authorization(
                target_data=target_data,
                target_validation=target_validation,
                casualty_assessment=casualty_assessment,
                operator=weapon_system.assigned_operator
            )
            
            if not human_authorization.approved:
                return TargetEngagementResult(
                    status='DENIED',
                    reason='Human authorization required but not granted',
                    audit_trail=self.create_audit_record(
                        target_data, target_validation, human_authorization
                    )
                )
        
        # Rules of engagement compliance check
        roe_check = self.weapons_oversight.verify_roe_compliance(
            target_data=target_data,
            engagement_type='lethal' if weapon_system.is_lethal else 'non_lethal',
            rules_of_engagement=weapon_system.current_roe
        )
        
        # Execute engagement with full audit trail
        if all([target_validation.valid, human_authorization.approved, roe_check.compliant]):
            engagement_result = self.execute_engagement(
                target_data=target_data,
                weapon_system=weapon_system,
                authorization=human_authorization
            )
            
            # Post-engagement assessment
            post_assessment = self.conduct_post_engagement_assessment(
                engagement_result=engagement_result,
                target_data=target_data,
                casualty_assessment=casualty_assessment
            )
            
            return TargetEngagementResult(
                status='EXECUTED',
                engagement_result=engagement_result,
                post_assessment=post_assessment,
                audit_trail=self.create_comprehensive_audit_record(
                    target_data, target_validation, human_authorization, 
                    engagement_result, post_assessment
                )
            )
        else:
            return TargetEngagementResult(
                status='ABORTED',
                reason='Failed compliance checks',
                failed_checks={
                    'target_validation': target_validation.valid,
                    'human_authorization': human_authorization.approved,
                    'roe_compliance': roe_check.compliant
                }
            )
