from ciaf.energy import EnergyAIGovernanceFramework

class EnergyAISystem:
    def __init__(self):
        self.governance = EnergyAIGovernanceFramework()
        
    def optimize_smart_grid(self, grid_data):
        """Optimize smart grid with AI governance"""
        
        # Privacy-preserving demand analysis
        demand_analysis = self.governance.analyze_demand_patterns(
            grid_data=grid_data,
            privacy_protection=True
        )
        
        # Renewable energy integration
        renewable_optimization = self.governance.optimize_renewable_integration(
            demand_analysis=demand_analysis,
            weather_data=grid_data.weather_forecast
        )
        
        return GridOptimizationResult(
            demand_prediction=demand_analysis,
            renewable_integration=renewable_optimization,
            efficiency_improvement=self.calculate_efficiency_gains()
        )
