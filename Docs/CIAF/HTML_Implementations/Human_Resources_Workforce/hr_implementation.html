from ciaf.human_resources import HRAIGovernanceFramework
from ciaf.bias import NYCLocalLaw144Compliance

class HRAISystem:
    def __init__(self):
        self.governance = HRAIGovernanceFramework()
        self.nyc_compliance = NYCLocalLaw144Compliance()
        
    def screen_candidates(self, candidates, job_requirements):
        """Screen candidates with bias prevention and fairness monitoring"""
        
        # NYC Local Law 144 compliance check
        compliance_check = self.nyc_compliance.validate_screening_tool(
            screening_algorithm=self.candidate_screening_model,
            job_requirements=job_requirements
        )
        
        # Bias detection across protected attributes
        bias_assessment = self.governance.assess_screening_bias(
            candidates=candidates,
            protected_attributes=['gender', 'race', 'age', 'disability'],
            screening_results=self.screen_candidates_internal(candidates)
        )
        
        # Fair hiring recommendations
        fair_recommendations = self.governance.generate_fair_recommendations(
            candidates=candidates,
            bias_assessment=bias_assessment,
            diversity_targets=job_requirements.diversity_goals
        )
        
        return HiringRecommendations(
            candidate_scores=fair_recommendations.scores,
            bias_assessment=bias_assessment,
            compliance_status=compliance_check,
            fairness_metrics=self.calculate_fairness_metrics(bias_assessment)
        )
