from ciaf.insurance import InsuranceAIGovernanceFramework
from ciaf.actuarial import ActuarialFairnessValidator

class InsuranceAISystem:
    def __init__(self):
        self.governance = InsuranceAIGovernanceFramework()
        self.fairness_validator = ActuarialFairnessValidator()
        
    def evaluate_risk_and_pricing(self, applicant_data, policy_type):
        """Evaluate insurance risk with actuarial fairness and bias prevention"""
        
        # Protected attribute identification
        protected_attrs = self.governance.identify_protected_attributes(
            applicant_data=applicant_data,
            regulatory_framework='NAIC_MODEL_LAWS'
        )
        
        # Fair risk assessment
        risk_assessment = self.governance.assess_risk_fairly(
            applicant_data=applicant_data,
            protected_attributes=protected_attrs,
            policy_type=policy_type
        )
        
        # Actuarial fairness validation
        fairness_check = self.fairness_validator.validate_pricing_fairness(
            risk_score=risk_assessment.risk_score,
            premium_calculation=risk_assessment.premium,
            demographic_group=applicant_data.demographics,
            comparable_groups=self.get_comparable_demographic_groups(applicant_data)
        )
        
        # Transparent pricing explanation
        pricing_explanation = self.governance.generate_pricing_explanation(
            risk_factors=risk_assessment.risk_factors,
            premium_components=risk_assessment.premium_breakdown,
            fairness_metrics=fairness_check.fairness_metrics
        )
        
        return InsurancePricingDecision(
            risk_score=risk_assessment.risk_score,
            premium=risk_assessment.premium,
            fairness_validation=fairness_check,
            explanation=pricing_explanation,
            compliance_status=self.verify_regulatory_compliance(risk_assessment)
        )
        
    def process_claims_fairly(self, claim_data):
        """Process insurance claims with bias prevention and fraud detection"""
        
        # Fraud detection with fairness constraints
        fraud_assessment = self.governance.detect_fraud_fairly(
            claim_data=claim_data,
            demographic_factors=claim_data.demographics,
            historical_patterns=self.get_historical_claim_patterns()
        )
        
        # Fair settlement calculation
        settlement = self.governance.calculate_fair_settlement(
            claim_data=claim_data,
            fraud_risk=fraud_assessment.fraud_probability,
            policy_terms=claim_data.policy_details
        )
        
        return ClaimsDecision(
            approval_status=settlement.approval,
            settlement_amount=settlement.amount,
            fraud_assessment=fraud_assessment,
            decision_explanation=settlement.explanation
        )
